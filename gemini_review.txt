class BrainV2:
    def __init__(self):
        self.playwright = None
        self.browser = None
        self.models = {}  # 结构: { "ModelName": page_instance }

    async def connect_and_discover(self):
        """强行建立异步 CDP 连接，动态扫描并建立大模型通信总线"""
        self.playwright = await async_playwright().start()
        try:
            self.browser = await self.playwright.chromium.connect_over_cdp("http://127.0.0.1:9222")
            self.refresh_discover()
        except Exception as e:
            raise RuntimeError(f"[BrainV2] CDP (9222) 端口劫持失败，请确保 Chrome 已带调试端口拉起。错误: {str(e)}")

    def refresh_discover(self):
        """高泛化存量扫描 + 动态增量感知监听（防页面刷新句柄丢失）"""
        if not self.browser or not self.browser.contexts:
            return
        context = self.browser.contexts[0]
        
        # 扫描存量标签
        for pg in context.pages:
            self._register_page_fingerprint(pg)
            
        # 监听中途新开或刷新的标签页
        context.on("page", lambda new_page: self._register_page_fingerprint(new_page))
